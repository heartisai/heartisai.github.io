$.formatDate(date, ‘yyyy-MM-dd HH:mm’); 2014-12-13 15:30
$.formatDate(date, ‘HH:mm:ss’); 15:30:12
$.formatDate(date, ‘yyyy年MM月dd日’); 2014年12月13日
